class KVExpSetUp : public KVMultiDetArray {

   KVString lmultidetarrayclasses;

   void init();

public:

   KVExpSetUp();
   virtual ~KVExpSetUp();
   virtual void Build(Int_t run = -1);

   void SetParameters(UShort_t n);

   }
   virtual void AcceptParticleForAnalysis(KVReconstructedNucleus*) const;

};
